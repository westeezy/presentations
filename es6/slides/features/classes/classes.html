class TodoModel {
    constructor(storage) {
        this.storage = storage;
    }

    create(title) {
        title = title || '';

        var newItem = {
            title: title.trim(),
            completed: false
        };

        return this.storage.save(newItem);
    }
}

export default TodoModel;
