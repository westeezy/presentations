Tagged Template Strings

Modify template strings with a function 


function tag(array_of_strings, ...substitions) { //get to rest param laster
	console.log(array_of_strings[0]); //Hello
	console.log(substitions[0]); //Westin
	console.log(array_of_strings[1]); //from js
}

var name = 'Westin';
tag`Hello ${name} from js`;
