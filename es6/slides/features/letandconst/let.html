Scope Binding constructs
  • In the early harmony days we (the community) tried to use try/catch to do this. Which was very imperformant
  • will not hoist

function() {
	if(x) {
		var foo = 3;
	}
	var baz = 1;

	//foo and baz in same scope due to hoisting
}


function() {
	if(x) {
		let foo = 3;
	}
	var baz = 1;

	//foo and baz in same scope due to hoisting
}
